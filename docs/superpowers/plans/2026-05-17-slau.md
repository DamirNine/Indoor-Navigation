export type SegmentType = 'number' | 'variable' | 'parameter' | 'operator' | 'equals' | 'whitespace' | 'unknown';

// One token in an equation string
export interface Segment {
  type: SegmentType;
  value: string;        // raw text (e.g. "x", "3", "+", "(X_m)")
  id: string;           // stable UUID, assigned on first parse
  sub: string;          // subscript, e.g. "12" — only for variable/parameter
  sup: string;          // superscript — only for variable/parameter
}

// One equation row
export interface EquationRow {
  id: string;
  segments: Segment[];  // parsed and annotated
  rawText: string;      // what the user typed, kept in sync with segments
}

// One Gauss elimination step
export interface GaussStep {
  desc: string;         // e.g. "R2 ← R2 − (4/3)·R1"
  matrix: string[][];   // augmented matrix after this step (fraction strings)
}

// Result of solving the system
export interface SolveResult {
  status: 'ok' | 'underdetermined' | 'inconsistent' | 'too_many_vars' | 'parse_error';
  errorMessage?: string;
  variables: string[];        // ordered list of unknown names (display form)
  answer: Record<string, string>;  // variable display name → value/expression string
  // only present when status === 'ok':
  initialMatrix?: string[][];  // augmented matrix before elimination
  steps?: GaussStep[];
  backSubstitution?: string[]; // lines like "x₃ = 1 → x₂₁ = -2 → x₁₂ = 2"
  // x₁₂/x₂₁ pairs detected:
  mirroredPairs?: Array<{ canonical: string; mirrored: string }>;
}

// History entry
export interface HistoryEntry {
  id: string;
  timestamp: number;
  equations: EquationRow[];
  result: SolveResult;
  preview: string;      // first equation rawText truncated to 40 chars
}

// src/parser/tokenizer.ts
import { Segment, SegmentType } from '../types'

// Unicode letter: latin + greek (U+0370–U+03FF) + extended greek (U+1F00–U+1FFF)
const LETTER = /[a-zA-ZͰ-Ͽἀ-῿]/
const LETTER_OR_DIGIT = /[a-zA-ZͰ-Ͽἀ-῿0-9_]/

function makeId() {
  return Math.random().toString(36).slice(2, 10)
}

export function tokenize(input: string): Segment[] {
  const segments: Segment[] = []
  let i = 0

  while (i < input.length) {
    const ch = input[i]

    // Whitespace
    if (/\s/.test(ch)) {
      let j = i
      while (j < input.length && /\s/.test(input[j])) j++
      segments.push({ type: 'whitespace', value: input.slice(i, j), id: makeId(), sub: '', sup: '' })
      i = j
      continue
    }

    // Parameter: (...) — nested parens handled
    if (ch === '(') {
      let j = i + 1, depth = 1
      while (j < input.length && depth > 0) {
        if (input[j] === '(') depth++
        if (input[j] === ')') depth--
        j++
      }
      segments.push({ type: 'parameter', value: input.slice(i, j), id: makeId(), sub: '', sup: '' })
      i = j
      continue
    }

    // Number: digits + optional dot
    if (/[0-9]/.test(ch)) {
      let j = i
      while (j < input.length && /[0-9.]/.test(input[j])) j++
      segments.push({ type: 'number', value: input.slice(i, j), id: makeId(), sub: '', sup: '' })
      i = j
      continue
    }

    // Variable: starts with a letter
    if (LETTER.test(ch)) {
      let j = i
      while (j < input.length && LETTER_OR_DIGIT.test(input[j])) j++
      segments.push({ type: 'variable', value: input.slice(i, j), id: makeId(), sub: '', sup: '' })
      i = j
      continue
    }

    // Equals
    if (ch === '=') {
      segments.push({ type: 'equals', value: '=', id: makeId(), sub: '', sup: '' })
      i++
      continue
    }

    // Operators
    if (/[+\-*/]/.test(ch)) {
      segments.push({ type: 'operator', value: ch, id: makeId(), sub: '', sup: '' })
      i++
      continue
    }

    // Unknown character: preserve as-is
    segments.push({ type: 'unknown', value: ch, id: makeId(), sub: '', sup: '' })
    i++
  }

  return segments
}

// Merge freshly-tokenized segments with stored segments to preserve sub/sup indices.
// Matching strategy: match by value+type in order. If same variable appears multiple times,
// match by occurrence index.
export function mergeIndices(
  fresh: Segment[],
  stored: Segment[]
): Segment[] {
  // Build occurrence map from stored: "type:value" → [id, sub, sup][]
  const occurrenceMap = new Map<string, Array<{ id: string; sub: string; sup: string }>>()
  for (const seg of stored) {
    if (seg.type !== 'variable' && seg.type !== 'parameter') continue
    const key = `${seg.type}:${seg.value}`
    if (!occurrenceMap.has(key)) occurrenceMap.set(key, [])
    occurrenceMap.get(key)!.push({ id: seg.id, sub: seg.sub, sup: seg.sup })
  }

  const usedCount = new Map<string, number>()

  return fresh.map(seg => {
    if (seg.type !== 'variable' && seg.type !== 'parameter') return seg
    const key = `${seg.type}:${seg.value}`
    const count = usedCount.get(key) ?? 0
    usedCount.set(key, count + 1)
    const stored_list = occurrenceMap.get(key)
    if (stored_list && stored_list[count]) {
      const s = stored_list[count]
      return { ...seg, id: s.id, sub: s.sub, sup: s.sup }
    }
    return seg
  })
}

// src/parser/normalizer.ts
import { EquationRow, Segment } from '../types'

// Display name for a variable segment
export function displayName(seg: { value: string; sub: string; sup: string }): string {
  let name = seg.value
  if (seg.sub) name += toSubscript(seg.sub)
  if (seg.sup) name += toSuperscript(seg.sup)
  return name
}

function toSubscript(s: string): string {
  const map: Record<string, string> = { '0':'₀','1':'₁','2':'₂','3':'₃','4':'₄','5':'₅','6':'₆','7':'₇','8':'₈','9':'₉' }
  return s.split('').map(c => map[c] ?? c).join('')
}

function toSuperscript(s: string): string {
  const map: Record<string, string> = { '0':'⁰','1':'¹','2':'²','3':'³','4':'⁴','5':'⁵','6':'⁶','7':'⁷','8':'⁸','9':'⁹' }
  return s.split('').map(c => map[c] ?? c).join('')
}

interface VarKey { name: string; sub: string; sup: string }

export function extractVariables(rows: EquationRow[]): string[] {
  const seen = new Set<string>()
  const result: string[] = []
  for (const row of rows) {
    for (const seg of row.segments) {
      if (seg.type !== 'variable') continue
      const dn = displayName(seg)
      if (!seen.has(dn)) { seen.add(dn); result.push(dn) }
    }
  }
  return result
}

export function detectMirroredPairs(
  vars: VarKey[]
): Array<{ canonical: VarKey; mirrored: VarKey }> {
  const pairs: Array<{ canonical: VarKey; mirrored: VarKey }> = []
  for (let i = 0; i < vars.length; i++) {
    for (let j = i + 1; j < vars.length; j++) {
      const a = vars[i], b = vars[j]
      if (a.name !== b.name || a.sup !== b.sup) continue
      if (a.sub.length < 2) continue // single digit: not a pair
      if (a.sub === b.sub.split('').reverse().join('')) {
        // canonical = lexicographically smaller sub
        if (a.sub < b.sub) pairs.push({ canonical: a, mirrored: b })
        else pairs.push({ canonical: b, mirrored: a })
      }
    }
  }
  return pairs
}

export interface LinearEquation {
  coefficients: Record<string, number>  // variable displayName → coefficient
  rhs: number
}

// Parse one row into a LinearEquation coefficient map.
// Mirrored variables (x₂₁ → -x₁₂) are folded into canonical.
export function normalizeForSolver(rows: EquationRow[]): {
  equations: LinearEquation[]
  variables: string[]
  pairs: Array<{ canonical: string; mirrored: string }>
} {
  // Collect all variable keys
  const varKeys: VarKey[] = []
  const varKeySeen = new Set<string>()
  for (const row of rows) {
    for (const seg of row.segments) {
      if (seg.type !== 'variable') continue
      const key = `${seg.value}|${seg.sub}|${seg.sup}`
      if (!varKeySeen.has(key)) {
        varKeySeen.add(key)
        varKeys.push({ name: seg.value, sub: seg.sub, sup: seg.sup })
      }
    }
  }

  const mirrorPairs = detectMirroredPairs(varKeys)
  // Build mirror map: mirroredDisplayName → { canonicalDisplayName, sign: -1 }
  const mirrorMap = new Map<string, { canonical: string; sign: number }>()
  for (const p of mirrorPairs) {
    mirrorMap.set(displayName(p.mirrored), { canonical: displayName(p.canonical), sign: -1 })
  }

  // Variables to solve for (canonical only)
  const allVarNames = varKeys.map(v => displayName(v))
  const variables = allVarNames.filter(v => !mirrorMap.has(v))

  // Parse equations
  const equations: LinearEquation[] = rows.map(row => parseRow(row, variables, mirrorMap))

  return {
    equations,
    variables,
    pairs: mirrorPairs.map(p => ({ canonical: displayName(p.canonical), mirrored: displayName(p.mirrored) })),
  }
}

function parseRow(
  row: EquationRow,
  variables: string[],
  mirrorMap: Map<string, { canonical: string; sign: number }>
): LinearEquation {
  const coefficients: Record<string, number> = {}
  for (const v of variables) coefficients[v] = 0

  const segs = row.segments.filter(s => s.type !== 'whitespace')
  let sign = 1
  let pendingCoeff: number | null = null
  let pendingParam: string | null = null
  let rhs = 0
  let onRhs = false

  for (let i = 0; i < segs.length; i++) {
    const seg = segs[i]

    if (seg.type === 'equals') { onRhs = true; sign = 1; continue }

    if (seg.type === 'operator') {
      sign = seg.value === '-' ? -1 : 1
      pendingCoeff = null
      pendingParam = null
      continue
    }

    if (seg.type === 'number') {
      pendingCoeff = parseFloat(seg.value) * sign
      continue
    }

    if (seg.type === 'parameter') {
      pendingParam = seg.value
      continue
    }

    if (seg.type === 'variable') {
      const dn = displayName(seg)
      const coeff = pendingCoeff ?? sign
      if (onRhs) {
        rhs -= coeff
      } else {
        const mirror = mirrorMap.get(dn)
        const target = mirror ? mirror.canonical : dn
        const effectiveCoeff = mirror ? coeff * mirror.sign : coeff
        if (target in coefficients) coefficients[target] += effectiveCoeff
      }
      pendingCoeff = null
      pendingParam = null
      sign = 1
      continue
    }
  }

  return { coefficients, rhs }
}

// src/solver/numeric.ts
import Fraction from 'fraction.js'
import { LinearEquation } from '../src/parser/normalizer'
import { GaussStep, SolveResult } from '../types'

type FMatrix = Fraction[][]

function cloneMatrix(m: FMatrix): string[][] {
  return m.map(row => row.map(f => f.toFraction(true)))
}

export function solveNumeric(
  equations: LinearEquation[],
  variables: string[]
): SolveResult {
  const n = variables.length
  const m = equations.length

  if (n > 20) {
    return { status: 'too_many_vars', variables, answer: {}, errorMessage: `Слишком много переменных: ${n} (максимум 20)` }
  }
  if (m < n) {
    return { status: 'underdetermined', variables, answer: {}, errorMessage: `Уравнений (${m}) меньше чем переменных (${n}). Система недоопределена.` }
  }

  // Build augmented matrix
  const mat: FMatrix = equations.map(eq =>
    [...variables.map(v => new Fraction(eq.coefficients[v] ?? 0)), new Fraction(eq.rhs)]
  )

  const initialMatrix = cloneMatrix(mat)
  const steps: GaussStep[] = []
  const cols = n + 1

  // Forward elimination with partial pivoting
  for (let col = 0; col < n; col++) {
    // Find pivot row
    let pivotRow = -1
    let maxVal = new Fraction(0)
    for (let row = col; row < m; row++) {
      const abs = mat[row][col].abs()
      if (abs.compare(maxVal) > 0) { maxVal = abs; pivotRow = row }
    }

    if (pivotRow === -1 || maxVal.equals(0)) continue // zero column

    if (pivotRow !== col) {
      ;[mat[col], mat[pivotRow]] = [mat[pivotRow], mat[col]]
      steps.push({ desc: `R${col+1} ↔ R${pivotRow+1}`, matrix: cloneMatrix(mat) })
    }

    for (let row = col + 1; row < m; row++) {
      if (mat[row][col].equals(0)) continue
      const factor = mat[row][col].div(mat[col][col])
      for (let j = 0; j < cols; j++) {
        mat[row][j] = mat[row][j].sub(factor.mul(mat[col][j]))
      }
      steps.push({
        desc: `R${row+1} ← R${row+1} − (${factor.toFraction(true)})·R${col+1}`,
        matrix: cloneMatrix(mat),
      })
    }
  }

  // Check consistency: any row [0 0 ... 0 | c] with c ≠ 0 → inconsistent
  for (let row = n; row < m; row++) {
    const allZero = variables.every((_, j) => mat[row][j].equals(0))
    if (allZero && !mat[row][n].equals(0)) {
      return {
        status: 'inconsistent',
        variables,
        answer: {},
        initialMatrix,
        steps,
        errorMessage: 'Система несовместна (нет решений).',
      }
    }
  }

  // Back substitution
  const solution: Fraction[] = new Array(n).fill(new Fraction(0))
  const backLines: string[] = []

  for (let row = n - 1; row >= 0; row--) {
    if (mat[row][row].equals(0)) continue
    let val = mat[row][n]
    for (let j = row + 1; j < n; j++) {
      val = val.sub(mat[row][j].mul(solution[j]))
    }
    solution[row] = val.div(mat[row][row])
    backLines.push(`${variables[row]} = ${solution[row].toFraction(true)}`)
  }

  const answer: Record<string, string> = {}
  for (let i = 0; i < n; i++) {
    answer[variables[i]] = solution[i].toFraction(true)
  }

  return {
    status: 'ok',
    variables,
    answer,
    initialMatrix,
    steps,
    backSubstitution: backLines,
  }
}

```typescript
// src/solver/symbolic.ts
// @ts-ignore — nerdamer has no bundled types
import nerdamer from 'nerdamer'
import 'nerdamer/Solve'
import { EquationRow } from '../types'
import { SolveResult } from '../types'
import { displayName } from '../parser/normalizer'

// Convert display name like "x₁₂" to nerdamer-safe ascii var name
function toNerdVar(display: string): string {
  return 'v' + display.split('').map(c => c.charCodeAt(0)).join('_')
}

// Build a nerdamer equation string from a row's segments
function buildNerdEq(
  row: EquationRow,
  varMap: Map<string, string>,  // display → nerd
  paramMap: Map<string, string> // "(expr)" → nerd param name
): string {
  const segs = row.segments.filter(s => s.type !== 'whitespace')
  let lhs = '', rhs = '', side: 'lhs'|'rhs' = 'lhs'

  for (const seg of segs) {
    if (seg.type === 'equals') { side = 'rhs'; continue }

    const target = side === 'lhs' ? (lhs += '') : (rhs += '')
    const append = (s: string) => { if (side === 'lhs') lhs += s; else rhs += s }

    if (seg.type === 'operator' || seg.type === 'number') {
      append(seg.value)
    } else if (seg.type === 'variable') {
      const dn = displayName(seg)
      append(varMap.get(dn) ?? dn)
    } else if (seg.type === 'parameter') {
      const pname = paramMap.get(seg.value) ?? seg.value.slice(1, -1).replace(/[^a-zA-Z0-9]/g, '_')
      append(pname)
    }
  }

  return `${lhs}=${rhs}`
}

// Back-convert nerdamer result string to display form
function fromNerdResult(result: string, reverseVarMap: Map<string, string>, reverseParamMap: Map<string, string>): string {
  let s = result
  for (const [nerd, display] of reverseVarMap) s = s.replaceAll(nerd, display)
  for (const [nerd, display] of reverseParamMap) s = s.replaceAll(nerd, display)
  return s
}

export function solveSymbolic(rows: EquationRow[], variables: string[]): SolveResult {
  // Build variable maps
  const varMap = new Map<string, string>()
  const reverseVarMap = new Map<string, string>()
  for (const v of variables) {
    const nv = toNerdVar(v)
    varMap.set(v, nv)
    reverseVarMap.set(nv, v)
  }

  // Collect parameters
  const paramMap = new Map<string, string>()
  const reverseParamMap = new Map<string, string>()
  let paramIdx = 0
  for (const row of rows) {
    for (const seg of row.segments) {
      if (seg.type !== 'parameter') continue
      if (!paramMap.has(seg.value)) {
        const pname = `p${paramIdx++}`
        const displayParam = seg.value.slice(1, -1).replace(/\s/g, '')
        paramMap.set(seg.value, pname)
        reverseParamMap.set(pname, displayParam)
      }
    }
  }

  // Build equation strings
  const eqStrings = rows.map(row => buildNerdEq(row, varMap, paramMap))
  const nerdVars = variables.map(v => varMap.get(v)!)

  try {
    const solutions = nerdamer.solveEquations(eqStrings, nerdVars)
    const answer: Record<string, string> = {}
    for (let i = 0; i < variables.length; i++) {
      const raw = Array.isArray(solutions) ? String(solutions[i] ?? '') : String(solutions)
      answer[variables[i]] = fromNerdResult(raw, reverseVarMap, reverseParamMap)
    }
    return { status: 'ok', variables, answer }
  } catch (e: any) {
    return { status: 'parse_error', variables, answer: {}, errorMessage: String(e.message ?? e) }
  }
}

// src/store.ts
import { create } from 'zustand'
import { EquationRow, HistoryEntry, SolveResult } from './types'
import { tokenize, mergeIndices } from './parser/tokenizer'

const STORAGE_KEY = 'slau_equations'
const HISTORY_KEY = 'slau_history'
const UNDO_LIMIT = 50

function makeId() {
  return Math.random().toString(36).slice(2, 10)
}

function emptyRow(): EquationRow {
  return { id: makeId(), rawText: '', segments: [] }
}

function loadEquations(): EquationRow[] {
  try {
    const raw = localStorage.getItem(STORAGE_KEY)
    if (raw) return JSON.parse(raw)
  } catch {}
  return [emptyRow()]
}

function loadHistory(): HistoryEntry[] {
  try {
    const raw = localStorage.getItem(HISTORY_KEY)
    if (raw) return JSON.parse(raw)
  } catch {}
  return []
}

interface State {
  equations: EquationRow[]
  past: EquationRow[][]   // undo stack
  future: EquationRow[][] // redo stack
  history: HistoryEntry[]
  symbolsPanelOpen: boolean
  historyPanelOpen: boolean

  // Actions
  updateRow: (id: string, rawText: string) => void
  setSegmentIndex: (rowId: string, segId: string, sub: string, sup: string) => void
  addRow: () => void
  removeRow: (id: string) => void
  clearAll: () => void
  undo: () => void
  redo: () => void
  pushHistory: (result: SolveResult) => void
  loadFromHistory: (entry: HistoryEntry) => void
  toggleSymbolsPanel: () => void
  toggleHistoryPanel: () => void
}

export const useStore = create<State>((set, get) => ({
  equations: loadEquations(),
  past: [],
  future: [],
  history: loadHistory(),
  symbolsPanelOpen: false,
  historyPanelOpen: false,

  updateRow(id, rawText) {
    const { equations, past } = get()
    const row = equations.find(r => r.id === id)
    if (!row) return
    const fresh = tokenize(rawText)
    const merged = mergeIndices(fresh, row.segments)
    const next = equations.map(r => r.id === id ? { ...r, rawText, segments: merged } : r)
    localStorage.setItem(STORAGE_KEY, JSON.stringify(next))
    set({ equations: next, past: [...past.slice(-UNDO_LIMIT + 1), equations], future: [] })
  },

  setSegmentIndex(rowId, segId, sub, sup) {
    const { equations, past } = get()
    const next = equations.map(row => {
      if (row.id !== rowId) return row
      return {
        ...row,
        segments: row.segments.map(s => s.id === segId ? { ...s, sub, sup } : s),
      }
    })
    localStorage.setItem(STORAGE_KEY, JSON.stringify(next))
    set({ equations: next, past: [...past.slice(-UNDO_LIMIT + 1), equations], future: [] })
  },

  addRow() {
    const { equations, past } = get()
    const next = [...equations, emptyRow()]
    localStorage.setItem(STORAGE_KEY, JSON.stringify(next))
    set({ equations: next, past: [...past.slice(-UNDO_LIMIT + 1), equations], future: [] })
  },

  removeRow(id) {
    const { equations, past } = get()
    const next = equations.length > 1 ? equations.filter(r => r.id !== id) : [emptyRow()]
    localStorage.setItem(STORAGE_KEY, JSON.stringify(next))
    set({ equations: next, past: [...past.slice(-UNDO_LIMIT + 1), equations], future: [] })
  },

  clearAll() {
    const { past, equations } = get()
    const next = [emptyRow()]
    localStorage.setItem(STORAGE_KEY, JSON.stringify(next))
    set({ equations: next, past: [...past.slice(-UNDO_LIMIT + 1), equations], future: [] })
  },

  undo() {
    const { past, equations, future } = get()
    if (!past.length) return
    const prev = past[past.length - 1]
    localStorage.setItem(STORAGE_KEY, JSON.stringify(prev))
    set({ equations: prev, past: past.slice(0, -1), future: [equations, ...future.slice(0, UNDO_LIMIT - 1)] })
  },

  redo() {
    const { past, equations, future } = get()
    if (!future.length) return
    const next = future[0]
    localStorage.setItem(STORAGE_KEY, JSON.stringify(next))
    set({ equations: next, past: [...past.slice(-UNDO_LIMIT + 1), equations], future: future.slice(1) })
  },

  pushHistory(result) {
    const { equations, history } = get()
    const entry: HistoryEntry = {
      id: makeId(),
      timestamp: Date.now(),
      equations: JSON.parse(JSON.stringify(equations)),
      result,
      preview: equations[0]?.rawText?.slice(0, 40) ?? '',
    }
    const next = [entry, ...history].slice(0, 10)
    localStorage.setItem(HISTORY_KEY, JSON.stringify(next))
    set({ history: next })
  },

  loadFromHistory(entry) {
    const { past, equations } = get()
    localStorage.setItem(STORAGE_KEY, JSON.stringify(entry.equations))
    set({ equations: entry.equations, past: [...past.slice(-UNDO_LIMIT + 1), equations], future: [] })
  },

  toggleSymbolsPanel: () => set(s => ({ symbolsPanelOpen: !s.symbolsPanelOpen })),
  toggleHistoryPanel: () => set(s => ({ historyPanelOpen: !s.historyPanelOpen })),
}))
```

// src/components/ResultPanel.tsx
import { useState } from 'react'
import { SolveResult } from '../types'
import { SolutionSteps } from './SolutionSteps'

interface Props {
  result: SolveResult
  mirroredPairs: Array<{ canonical: string; mirrored: string }>
}

export function ResultPanel({ result, mirroredPairs }: Props) {
  const [showSteps, setShowSteps] = useState(false)

  const handleCopy = () => {
    const text = Object.entries(result.answer)
      .map(([k, v]) => `${k} = ${v}`)
      .join('\n')
    navigator.clipboard.writeText(text)
  }

  if (result.status !== 'ok') {
    return (
      <div style={{
        background:'#2a1a1a', border:'1.5px solid #7a2a2a', borderRadius:10,
        padding:'16px 20px', marginTop:16,
      }}>
        <div style={{ fontSize:14, color:'#ff6a6a', fontWeight:600, marginBottom:6 }}>
          {result.status === 'underdetermined' && 'Система недоопределена'}
          {result.status === 'inconsistent' && 'Система несовместна'}
          {result.status === 'too_many_vars' && 'Слишком много переменных'}
          {result.status === 'parse_error' && 'Ошибка разбора'}
        </div>
        <div style={{ fontSize:13, color:'#cc8888' }}>{result.errorMessage}</div>
      </div>
    )
  }

  return (
    <div style={{ marginTop:16 }}>
      {/* Кнопка toggle */}
      <button
        onClick={() => setShowSteps(s => !s)}

          color:'#a0a0cc', fontSize:13, padding:'5px 14px', cursor:'pointer', marginBottom:12 }}

        {showSteps ? '▲ Скрыть решение' : '▼ Показать решение'}
      </button>

            steps={result.steps}
            backSubstitution={result.backSubstitution ?? []}
          />
          <hr style={{ border:'none', borderTop:'1px solid #2a2a4a', margin:'16px 0' }} />
        </div>
      )}

      {/* Ответ */}
      <div style={{
        background:'#1a2a1a', border:'1.5px solid #2a5a2a', borderRadius:10,
        padding:'14px 18px',
      }}>

                </span>
              )}
            </div>
          )
        })}
      </div>
    </div>
  )

// src/App.tsx
import { useEffect, useRef, useState } from 'react'
import { useStore } from './store'
import { EquationRow } from './components/EquationRow'
import { Toolbar } from './components/Toolbar'
import { SymbolsPanel } from './components/SymbolsPanel'
import { HistoryPanel } from './components/HistoryPanel'
import { ResultPanel } from './components/ResultPanel'
import { normalizeForSolver, extractVariables } from './parser/normalizer'
import { solveNumeric } from './solver/numeric'
import { solveSymbolic } from './solver/symbolic'
import { SolveResult } from './types'

export default function App() {
  const { equations, undo, redo, past, future, pushHistory, symbolsPanelOpen } = useStore()
  const [result, setResult] = useState<SolveResult | null>(null)
  const [mirroredPairs, setMirroredPairs] = useState<Array<{ canonical: string; mirrored: string }>>([])
  const activeRowId = useRef<string | null>(null)

  // Keyboard shortcuts
  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      if ((e.ctrlKey || e.metaKey) && e.key === 'z' && !e.shiftKey) { e.preventDefault(); undo() }
      if ((e.ctrlKey || e.metaKey) && (e.key === 'y' || (e.key === 'z' && e.shiftKey))) { e.preventDefault(); redo() }
    }
    window.addEventListener('keydown', handler)
    return () => window.removeEventListener('keydown', handler)
  }, [undo, redo])

  const handleSolve = () => {
    const { equations: normalized, variables, pairs } = normalizeForSolver(equations)
    const hasParams = equations.some(row => row.segments.some(s => s.type === 'parameter'))

    let res: SolveResult
    if (hasParams) {
      // Symbolic solve: pass original rows + variable list
      const { solveSymbolic: _ } = require('./solver/symbolic')
      res = solveSymbolic(equations, variables)
    } else {
      res = solveNumeric(normalized, variables)
    }
    res.mirroredPairs = pairs

    setResult(res)
    setMirroredPairs(pairs)
    pushHistory(res)
  }

  const handleInsertSymbol = (symbol: string) => {
    const id = activeRowId.current ?? equations[equations.length - 1]?.id
    if (!id) return
    const row = equations.find(r => r.id === id)
    if (!row) return
    const { updateRow } = useStore.getState()
    updateRow(id, row.rawText + symbol)
  }

  return (
    <div style={{ maxWidth:900, margin:'0 auto', padding:'20px 16px', minHeight:'100vh' }}>
      <h1 style={{ fontSize:22, color:'#a0cfff', marginBottom:4 }}>Решатель СЛАУ</h1>
      <p style={{ fontSize:12, color:'#555', marginBottom:20 }}>
        Система линейных алгебраических уравнений · ТММ
      </p>

      <div style={{ display:'flex', gap:16, alignItems:'flex-start' }}>
        <div style={{ flex:1, minWidth:0 }}>
          <Toolbar
            onSolve={handleSolve}
            canUndo={past.length > 0}
            canRedo={future.length > 0}
          />

          {equations.map((row, i) => (
            <div key={row.id} onFocus={() => { activeRowId.current = row.id }}>
              <EquationRow row={row} index={i} />
            </div>
          ))}

          {result && (
            <ResultPanel result={result} mirroredPairs={mirroredPairs} />
          )}
        </div>

        {symbolsPanelOpen && (
          <SymbolsPanel onInsert={handleInsertSymbol} />
        )}
      </div>

      <HistoryPanel />
    </div>
  )
}
